Solution 2 – Average of Three Numbers

//Write a program to find average of 3 numbers.
#include 

int main(){
    int a,b,c;
    printf("Enter three numbers: ");
    scanf("%d %d %d", &a, &b, &c);
    float avg = (float)(a+b+c)/3;
    printf("Average of three numbers: %.2f", avg);
    return 0;
}
Solution 3 – Area of Rectangle and Square

//Write a program to calculate rectangle area, square area.
#include 

int main(){
    int l,w,s;
    printf("Enter the length and width of rectangle: ");
    scanf("%d %d", &l, &w);
    printf("Area of rectangle: %d\n", l*w);
    printf("Enter the side of square: ");
    scanf("%d", &s);
    printf("Area of square: %d", s*s);
    return 0;
}
